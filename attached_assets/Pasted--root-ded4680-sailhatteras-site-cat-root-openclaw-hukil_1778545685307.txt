"""
hukilau_client.py — OpenClaw Agent Secret Puller
Run on Danforth or Hope to pull secrets from Hukilau and write to .env

Usage:
    python hukilau_client.py --agent danforth
    python hukilau_client.py --agent hope --env-file C:\\Users\\jayph\\.openclaw\\.env

Add to Windows Task Scheduler or run at agent startup.
"""

import argparse
import json
import os
import sys
import requests
from pathlib import Path
from datetime import datetime

# --- CONFIG ---
HUKILAU_URL = "https://hatterassailing.org/hukilau/api/secrets.php"

# Store tokens in Windows Credential Manager or just set here
# Better: set as Windows env vars so this file doesn't hold secrets
AGENT_TOKENS = {
    "danforth": os.environ.get("HUKILAU_TOKEN_DANFORTH", "22cd237a76201f033357011862bda7d3f5995f90541dbcdbd3f22711c6b0464c"),
    "hope":     os.environ.get("HUKILAU_TOKEN_HOPE",     "5141a59f5e03385b3eb88f7400021f1ce904218df2b6550abc3e81ef7e86c5ee"),
    "hukilau":  os.environ.get("HUKILAU_TOKEN_HUKILAU", "22cd237a76201f033357011862bda7d3f5995f90541dbcdbd3f22711c6b0464c"),
}

DEFAULT_ENV_PATH = Path(os.environ.get("USERPROFILE", "~")).expanduser() / ".openclaw" / ".env"

# --- Logic ---
def pull_secrets(agent_id: str) -> dict:
    token = AGENT_TOKENS.get(agent_id)
    if not token or token.startswith("PASTE_"):
        print(f"[hukilau] ERROR: No token configured for agent '{agent_id}'")
        sys.exit(1)

    try:
        resp = requests.post(
            HUKILAU_URL,
            headers={"Authorization": f"Bearer {token}"},
            timeout=10,
            verify=True  # always verify SSL
        )
        resp.raise_for_status()
        return resp.json()
    except requests.exceptions.SSLError:
        print("[hukilau] ERROR: SSL verification failed — do not proceed without valid cert")
        sys.exit(1)
    except requests.exceptions.RequestException as e:
        print(f"[hukilau] ERROR: Could not reach Hukilau: {e}")
        sys.exit(1)

def write_env(secrets: dict, env_path: Path) -> None:
    env_path.parent.mkdir(parents=True, exist_ok=True)
    lines = [f"# Auto-generated by hukilau_client.py — {datetime.now().isoformat()}\n"]
    if isinstance(secrets, dict):
        for k, v in secrets.items():
            lines.append(f"{k}={v}\n")
    env_path.write_text("".join(lines), encoding="utf-8")
    print(f"[hukilau] Wrote {len(secrets)} secrets to {env_path}")

def set_process_env(secrets: dict) -> None:
    """Optionally inject into current process env (if called as import)"""
    for k, v in secrets.items():
        os.environ[k] = v

def main():
    parser = argparse.ArgumentParser(description="Pull secrets from Hukilau vault")
    parser.add_argument("--agent", required=True, choices=["danforth", "hope", "hukilau"], help="Agent ID")
    parser.add_argument("--env-file", default=str(DEFAULT_ENV_PATH), help="Path to write .env file")
    parser.add_argument("--stdout", action="store_true", help="Print secrets as JSON (no file write)")
    args = parser.parse_args()

    print(f"[hukilau] Pulling secrets for agent: {args.agent}")
    secrets = pull_secrets(args.agent)
    print(f"[hukilau] Received {len(secrets)} secrets")

    if args.stdout:
        # Mask values in output
        masked = {k: v[:4] + "****" for k, v in secrets.items()}
        print(json.dumps(masked, indent=2))
    else:
        write_env(secrets, Path(args.env_file))

    print("[hukilau] Done.")

if __name__ == "__main__":
    main()
